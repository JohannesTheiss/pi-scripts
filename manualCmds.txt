1. Change pw of pi user:

2. enable ssh-keys
    ssh-copy-id pi@192.168.2.117


3. run initScript.sh

4. check country code
    sudo cat /etc/wpa_supplicant/wpa_supplicant.conf
    - country should be "DE"



4. Set wifi up:
    - enable wifi
        - rfkill list 
        - if soft blocked then unblock wifi
            sudo rfkill unblock wifi

    - list wifi: 
        sudo iwlist wlan0 scan 

    - encrypt password: 
        sudo su
        wpa_passphrase "SSID" PASSWORD >> /etc/wpa_supplicant/wpa_supplicant.conf

    - delete plain text of password
        sudo vi /etc/wpa_supplicant/wpa_supplicant.conf
 
    - reconfigure the interface: 
        wpa_cli -i wlan0 reconfigure

    - add to /etc/network/interfaces (muss man nicht??)

cat << LINES >> /etc/network/interfaces 
auto lo
iface lo inet loopback

auto eth0
allow-hotplug eth0
iface eth0 inet static
address 192.168.2.187
netmask 255.255.255.0

auto wlan0
allow-hotplug wlan0
iface wlan0 inet static
wpa-conf /etc/wpa_supplicant/wpa_supplicant.conf
address 192.168.2.188
netmask 255.255.255.0
broadcast 192.168.2.255
gateway 192.168.2.1

iface default inet dhcp
LINES
